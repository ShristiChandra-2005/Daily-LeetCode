Approch
=========
Check if strings are equal: If yes, return true.
Check if lengths are different: If yes, return false.
Find indices where characters differ.
If there are exactly two differences, check if swapping the characters at those indices makes the strings equal.
Return true if swap is valid, else return false.

Code::--
class Solution {
    public boolean areAlmostEqual(String s1, String s2) {
       if(s1.equals(s2)){
        return true;
       }
       if(s1.length()!=s2.length()){
        return false;
       }
       List <Integer> diff=new ArrayList();
       for(int i=0;i<s1.length(); i++){
        if(s1.charAt(i) != s2.charAt(i)){
            diff.add(i);
        }
    }
    if(diff.size() != 2){
        return false;
    }
    return s1.charAt(diff.get(0)) == s2.charAt(diff.get(1)) && s1.charAt(diff.get(1))==s2.charAt(diff.get(0));
    }
}
